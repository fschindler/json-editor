## Typed `JsonForm` component

<div className="code code--inline">

```ts
import { Editor } from '@sagold/react-json-editor';
import { JsonForm } from '@sagold/rje-widgets';

type MyData = {
  title: string
};

function MyForm({ schema }: { schema: JsonSchema }) {
  const [editor, setEditor] = useState<Editor<MyData>>();
  // data: MyData
  const data = editor?.getData();
  return (
    <JsonForm<MyData>
      // (editor: JsonEditor<MyData>) => void)
      editor={setEditor}
      // (data: MyData) => void
      onChange={(data) => {}}
      schema={schema}
    />
  );
}
```
